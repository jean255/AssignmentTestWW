async function load() { showLoading(true, '데이터 불러오는 중...'); try { const res = await fetch(`${JSONBIN_URL}/latest`, { headers: { 'X-Master-Key': JSONBIN_KEY } }); const json = await res.json(); db = json.record; migrate(); if (db.members.length === 0) { db.members = SEED_MEMBERS.map((s, i) => ({ id: uid(), name: s.name, role: '', color: COLORS[i % COLORS.length], cats: s.cats, })); await save(); } localStorage.setItem('task-app-db', JSON.stringify(db)); } catch (e) { try { const saved = localStorage.getItem('task-app-db'); if (saved) { db = JSON.parse(saved); migrate(); } } catch (_) {} toast('⚠️ 서버 연결 실패 — 로컬 데이터로 표시합니다'); } showLoading(false); renderAll(); } async function save() { localStorage.setItem('task-app-db', JSON.stringify(db)); try { await fetch(JSONBIN_URL, { method: 'PUT', headers: { 'Content-Type': 'application/json', 'X-Master-Key': JSONBIN_KEY }, body: JSON.stringify(db) }); } catch (e) { toast('⚠️ 저장 실패 — 네트워크를 확인해주세요'); } } async function syncData() { showLoading(true, '최신 데이터 동기화 중...'); try { const res = await fetch(`${JSONBIN_URL}/latest`, { headers: { 'X-Master-Key': JSONBIN_KEY } }); const json = await res.json(); db = json.record; migrate(); localStorage.setItem('task-app-db', JSON.stringify(db)); toast('✅ 동기화 완료'); } catch (e) { toast('⚠️ 동기화 실패 — 네트워크를 확인해주세요'); } showLoading(false); renderAll(); }
